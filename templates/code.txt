from flask import Flask, render_template, request, redirect
from flask import jsonify, send_file
from flask_mysqldb import MySQL

import pandas as pd
import plotly
import plotly.express as px

import json
from io import StringIO

app = Flask(__name__)

# MYSQL CONFIGURATION

app.config['MYSQL_HOST'] = 'localhost'
app.config['MYSQL_USER'] = 'root'
app.config['MYSQL_PASSWORD'] = '7010517079'
app.config['MYSQL_DB'] = 'analytics_dashboard'

mysql = MySQL(app)

# HOME PAGE

@app.route('/')
def index():
    return render_template('index2.html')

# FILE UPLOAD

@app.route('/upload', methods=['POST'])
def upload():

    file = request.files['file']

    # filter
    region = request.args.get('region')
    print(region)

    # READ FILE

    if file.filename.endswith('.csv'):
        df = pd.read_csv(file)

    else:
        df = pd.read_excel(file)

    # DATA CLEANING

    df.drop_duplicates(inplace=True)

    df.fillna(0, inplace=True)

    # DATE FORMAT

    df['order_date'] = pd.to_datetime(df['order_date'],dayfirst=True,errors='coerce')

    cur = mysql.connection.cursor()

    # INSERT DATA

    for _, row in df.iterrows():

        cur.execute("""

        INSERT INTO sales_data
        (order_date, region, category,
        product, sales, quantity, profit)

        VALUES (%s,%s,%s,%s,%s,%s,%s)

        """, (

            row['order_date'],
            row['region'],
            row['category'],
            row['product'],
            row['sales'],
            row['quantity'],
            row['profit']

        ))

    mysql.connection.commit()

    cur.close()

    return redirect('/dashboard')

# DASHBOARD

@app.route('/dashboard')
def dashboard():

    region = request.args.get('region')

    query = "SELECT * FROM sales_data"

    if region and region != 'All':
        query += f" WHERE region='{region}'"

    df = pd.read_sql(query, mysql.connection)

    # KPIs

    total_sales = round(df['sales'].sum(), 2)

    total_profit = round(df['profit'].sum(), 2)

    total_orders = len(df)

    avg_sales = round(df['sales'].mean(), 2)

    # GET REGIONS

    regions = pd.read_sql(
        "SELECT DISTINCT region FROM sales_data",
        mysql.connection
    )

    # CHARTS

    bar_fig = px.bar(
        df,
        x='category',
        y='sales',
        color='category',
        title='Sales by Category'
    )

    pie_fig = px.pie(
        df,
        names='region',
        values='sales',
        title='Region Wise Sales'
    )

    line_fig = px.line(
        df,
        x='order_date',
        y='sales',
        title='Sales Trend'
    )

    area_fig = px.area(
        df,
        x='order_date',
        y='profit',
        title='Profit Analysis'
    )

    # JSON

    barJSON = json.dumps(
        bar_fig,
        cls=plotly.utils.PlotlyJSONEncoder
    )

    pieJSON = json.dumps(
        pie_fig,
        cls=plotly.utils.PlotlyJSONEncoder
    )

    lineJSON = json.dumps(
        line_fig,
        cls=plotly.utils.PlotlyJSONEncoder
    )

    areaJSON = json.dumps(
        area_fig,
        cls=plotly.utils.PlotlyJSONEncoder
    )
    return render_template(

        'dashboard.html',

        total_sales=total_sales,
        total_profit=total_profit,
        total_orders=total_orders,
        avg_sales=avg_sales,

        regions=regions['region'].tolist(),

        selected_region=region,

        barJSON=barJSON,
        pieJSON=pieJSON,
        lineJSON=lineJSON,
        areaJSON=areaJSON,
    )

# FILTER DATA

@app.route('/filter')
def filter_data():

    start = request.args.get('start')
    end = request.args.get('end')

    query = f"""

    SELECT * FROM sales_data

    WHERE order_date
    BETWEEN '{start}' AND '{end}'

    """

    df = pd.read_sql(query, mysql.connection)

    return jsonify(
        df.to_dict(orient='records')
    )

# EXPORT CSV

@app.route('/export')
def export_csv():

    df = pd.read_sql(
        "SELECT * FROM sales_data",
        mysql.connection
    )

    output = StringIO()

    df.to_csv(output, index=False)

    output.seek(0)

    return send_file(
        output,
        mimetype='text/csv',
        as_attachment=True,
        download_name='analytics_export.csv'
    )

# AUTO REFRESH API

@app.route('/refresh')
def refresh():

    df = pd.read_sql(
        "SELECT * FROM sales_data",
        mysql.connection
    )

    return jsonify({

        "sales": float(df['sales'].sum()),

        "profit": float(df['profit'].sum()),

        "orders": int(len(df))

    })

if __name__ == '__main__':
    app.run(debug=True)
